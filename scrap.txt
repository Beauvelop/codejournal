function translatePigLatin(str) {
  var vow = ['a', 'e', 'i', 'o','u' ];
  res = str.split('');

  if (vow.includes(str.charAt(0))){
    return str += 'way';
  } else {
  for (var i = 0; i < str.length; i++) {
    if (!vow.includes(str[i])){ //as long as it's a consonant
    res.push(res.shift());
    }else {
    res.push('ay');
    return res.join('');
    }
  }
  }
}
