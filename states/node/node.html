A session is something on the back end that is stored as an ID--associated to an object.
A cookie is the mechanism by which we pass that ID back and forth.

.sid = session id.
(express-session is built on top of Connect)

This is what gets tossed back and forth on each of our requests.

  

    
A sample GET request



var members = [
  {name: "Bill", age: 34, phone: "Pixel"},
  {name: "Sue", age: 12, phone: "Android"},
  {name: "Lex", age: 44, phone: "iPhone"},
  {name: "Amy", age: 52, phone: "Pixel"},
  {name: "Don", age: 84, phone: "Landline"}
];

// localhost:3000/members?phone=Pixel
app.get('/members', function(req, res, next) {
  var filteredMembers = members.filter(function(members) {
    return members.phone === req.query.phone;
  })
  res.status(200).json(filteredMembers);
});
